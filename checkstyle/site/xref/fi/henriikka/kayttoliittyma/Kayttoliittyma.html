1   package fi.henriikka.kayttoliittyma;
2   
3   import java.awt.Container;
4   import java.awt.Dimension;
5   import java.awt.GridLayout;
6   import javax.swing.JButton;
7   import javax.swing.JFrame;
8   import javax.swing.JPanel;
9   import javax.swing.JTextField;
10  import javax.swing.WindowConstants;
11  
12  /**
13   * Luokka luo laskimen graafisen käyttöliittymän.
14   */
15  public class Kayttoliittyma implements Runnable {
16  
17      private JFrame frame;
18  
19      /**
20       * Metodi luo kehyksen laskimelle, asettaa sille koon ja sulkuoperaation,
21       * sekä luo komponentit ja asettaa ne näkyviksi.
22       *
23       */
24      @Override
25      public void run() {
26          frame = new JFrame("Kalkulaattori");
27          frame.setPreferredSize(new Dimension(400, 250));
28          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
29  
30          luoKomponentit(frame.getContentPane());
31  
32          frame.pack();
33          frame.setVisible(true);
34  
35      }
36  
37      /**
38       * Metodi luo laskimen "layoutin", tekstikentät, nappulapaneelit ja nappulat
39       * Lisää tekstikentät containeriin, nappulat paneeleihinsa ja paneelit
40       * containeriin Nimeää ne tekstikentät ja nappulat, ja asettaa niille
41       * halutun toiminnallisuuden: tuloskenttää ei voi muokata, liittää
42       * tapahtumankuuntelijan jokaiseen
43       *
44       * @param container Laskimen container
45       *
46       */
47      private void luoKomponentit(Container container) {
48  
49          frame.setLayout(new GridLayout(5, 1));
50          
51  
52          JTextField syotekentta = new JTextField();
53          JTextField toinenSyotekentta = new JTextField();
54          JTextField tuloskentta = new JTextField();
55          
56          tuloskentta.setEnabled(false);
57  
58          container.add(syotekentta);
59          container.add(toinenSyotekentta);
60          container.add(tuloskentta);
61          
62  
63          JButton summa = new JButton("+");
64          JButton erotus = new JButton("-");
65          JButton tulo = new JButton("*");
66          JButton osamaara = new JButton("/");
67          JButton nollaa = new JButton("C");
68          JButton viimeisinVastaus = new JButton("A");
69          JButton pvm = new JButton("D");
70          JButton tuumatSenteiksi = new JButton("IC");
71          JButton sentitTuumiksi = new JButton("CI");
72          JButton kilotPaunoiksi = new JButton("KP");
73          JButton paunatKiloiksi = new JButton("PK");
74          
75  
76          tuloskentta.setName("tulos");
77          syotekentta.setName("syote");
78          toinenSyotekentta.setName("tsyote");
79          summa.setName("+");
80          erotus.setName("-");
81          tulo.setName("*");
82          osamaara.setName("/");
83          nollaa.setName("C");
84          viimeisinVastaus.setName("A");
85          pvm.setName("D");
86          tuumatSenteiksi.setName("IC");
87          sentitTuumiksi.setName("CI");
88          kilotPaunoiksi.setName("KP");
89          paunatKiloiksi.setName("PK");
90          
91  
92          Tapahtumankuuntelija kasittelija = new Tapahtumankuuntelija(
93                  summa, erotus, tulo, osamaara, nollaa, viimeisinVastaus,
94                  pvm, tuumatSenteiksi, sentitTuumiksi, kilotPaunoiksi, paunatKiloiksi,
95                  tuloskentta, syotekentta, toinenSyotekentta);
96  
97          summa.addActionListener(kasittelija);
98          erotus.addActionListener(kasittelija);
99          tulo.addActionListener(kasittelija);
100         osamaara.addActionListener(kasittelija);
101         nollaa.addActionListener(kasittelija);
102         viimeisinVastaus.addActionListener(kasittelija);
103         pvm.addActionListener(kasittelija);
104         tuumatSenteiksi.addActionListener(kasittelija);
105         sentitTuumiksi.addActionListener(kasittelija);
106         kilotPaunoiksi.addActionListener(kasittelija);
107         paunatKiloiksi.addActionListener(kasittelija);
108         
109 
110         JPanel nappulapaneeli = new JPanel(new GridLayout(1, 6));
111         nappulapaneeli.add(summa);
112         nappulapaneeli.add(erotus);
113         nappulapaneeli.add(tulo);
114         nappulapaneeli.add(osamaara);
115         nappulapaneeli.add(nollaa);
116         nappulapaneeli.add(viimeisinVastaus);
117         
118 
119         JPanel toinenNappulapaneeli = new JPanel(new GridLayout(1, 5));
120         toinenNappulapaneeli.add(pvm);
121         toinenNappulapaneeli.add(tuumatSenteiksi);
122         toinenNappulapaneeli.add(sentitTuumiksi);
123         toinenNappulapaneeli.add(kilotPaunoiksi);
124         toinenNappulapaneeli.add(paunatKiloiksi);
125         
126 
127         container.add(nappulapaneeli);
128         container.add(toinenNappulapaneeli);
129     }
130 
131     /**
132      * Metodi palauttaa laskimen kehyksen.
133      *
134      * @return kehys
135      */
136     public JFrame getFrame() {
137         return frame;
138     }
139 
140 }
