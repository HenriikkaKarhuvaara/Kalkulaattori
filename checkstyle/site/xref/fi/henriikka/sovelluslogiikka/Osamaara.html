3   /**
4    * Luokka tarjoaa metodin "laske" suorituksen toiminnolle osamaara.
5    */
6   public class Osamaara extends PerusToiminto {
7   
8       private boolean onnistuikoJako;
9   
10      public Osamaara() {
11          onnistuikoJako = true;
12      }
13  
14      /**
15       * Metodi laskee annettujen parametrien osamaaran, jos jakaja ei ole nolla,
16       * ja palauttaa sen arvon
17       * Asettaa sen yläluokan parametrin tulos arvoksi
18       * Jos jakaja on nolla, kertoo metodi, että
19       * jakaminen ei onnistu asettamalla booleanin onnistuikoJako arvoksi false.
20       *
21       * @param x Käyttäjän antama syöte
22       * @param y Käyttäjän antama syöte
23       *
24       * @return Yläluokan Toiminto parametrin tulos arvo, eli käyttäjän
25       * syötteiden osamaara.
26       */
27      @Override
28      public double suoritaToiminto(double x, double y) {
29  
30          if (y == 0) {
31              onnistuikoJako = false;
32          } else {
33              super.tulos = x / y;
34              return super.tulos;
35          }
36          return 0;
37      }
38  
39      /**
40       * Metodi kertoo, onko jakaminen onnistunut.
41       *
42       * @return Boolean true tai false sen mukaan, onko jako tehty.
43       */
44      public boolean onnistuikoJako() {
45          return this.onnistuikoJako;
46      }
47  }
