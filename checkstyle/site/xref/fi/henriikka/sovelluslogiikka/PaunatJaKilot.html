4   /**
5    * Luokka tarjoaa mahdollisuuden muuttaa kilot paunoiksi ja paunat kiloiksi.
6    * 
7    */
8   public class PaunatJaKilot {
9       private double kilot;
10      private double paunat;
11      
12      
13      /**
14       * Metodi tulkitsee käyttäjän syötteen paunoina, ja muuntaa sen kiloiksi.
15       * 
16       * @param paunat käyttäjän syöttämä paino
17       * 
18       * @return Käyttäjän syöte muunnettuna kiloiksi
19       */
20      public double paunatKiloiksi(double paunat) {
21          
22          this.kilot = paunat / 2.2046;
23          return this.kilot;
24          
25      }
26      
27       /**
28       * Metodi tulkitsee käyttäjän syötteen kiloina, ja muuntaa sen paunoiksi.
29       * 
30       * @param kilot käyttäjän söyttämä paino
31       * 
32       * @return Käyttäjän syöte muunnettuna paunoiksi
33       */
34      public double kilotPaunoiksi(double kilot) {
35          
36          this.paunat = kilot * 2.2046;
37          return this.paunat;
38      }
39  }
